model{
    # toy model for testing negative binomial minimum count
    #r ~ dunif(0,5000)
    
    p2 ~ dunif(0,1)
    for(i in 1:nyear){
      p1[i] ~ dunif(0,1)
      N[i] ~ dpois(500)
      M[i] ~ dpois(500)
    }
    for(i in 1:nyear){
      mindat1[i] ~ dbin(p1[i], N[i])
      mindat2[i] ~ dnegbin(p2, M[i])
    }
  }
